## 常用命令

#### 常用命令

```shell
# 版本信息
docker version
# 系统信息，包括镜像和容器的数量
docker info
# helo
docker [command] --help
```

#### 镜像命令

```shell
# 镜像列表
docker images -a

# 删除镜像
docker rmi [image-id]
# 删除多个镜像
docker rmi [image1-id] [image2-id] [image3-id]
# 删除全部
docker rmi -f $(docker images -aq)

# 搜索镜像
docker search [image-name]
# 过滤star数1000以上的
docker search [image-name] --filter=START=1000

# 下载镜像
docker pull
# 指定版本下载
docker pull [image-name]:[version]
```

#### 容器命令

```shell
# 显示容器
docker ps
		# 列出当前正在运行的容器
-a		# 列出当前正在运行的容器 + 历史运行过的容器
-n=[n]	# 列出最近的n的容器
-q		# 列出容器的编号
-aq		# 列出当前正在运行的容器 + 历史运行过的容器的编号

# 新建容器 
docker run [params] [image-name]

# 退出容器
# 停止容器并退出
exit	
# 退出容器不停止
Ctrl + P + Q	

# 删除容器
docker rm [container-id]
docker rm [container1-id] [container2-id] [container2-id]
docker rm -f $(docker ps -aq)


# 查看容器中最近的n条日志
docker logs -f -t --tail [n] [container-id]
# 查看容器中的进程信息
docker top [container-id]
# 查看容器的元数据
docker inspect [container-id]

# 进入当前正在运行的容器
docker exec -it [container-id] /bin/bash # 进入容器后开启一个新的终端
docker attach [container-id]			 # 进入容器正在执行的终端

# 拷贝容器内的文件(文件夹)到主机
docker cp [container-id]:[container-path] [host-path]
```
